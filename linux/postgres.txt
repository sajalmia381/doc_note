# sudo -i -u postgres

// Create database
  $ createdb 'databse_name'

// drop databse
  $ dropdb "databse_name"

// access databse
  $ psql "database_name"

// Switch Database
  $ \c 'database_name'

// Database List
  $ psql --list

* if authenticated psql databse

     $ \d

     $ \l

     $ DROP TABLE "table_name";

  // Access Table Data
